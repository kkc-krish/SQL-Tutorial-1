---EXAMPLE USING SYSTEM ERROR FUNCTIONS


BEGIN TRY 
		SELECT 5/0 
END TRY 
BEGIN CATCH 
	SELECT 
    ERROR_NUMBER() AS ErrNumber,      
    ERROR_SEVERITY() AS ErrSeverity,
    ERROR_STATE() AS ErrState, 
    ERROR_PROCEDURE() AS ErrProc, 
    ERROR_LINE() AS ErrLine, 
    ERROR_MESSAGE() AS ErrMessage 
END CATCH
